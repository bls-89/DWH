1)SELECT firstname,lastname  
FROM orders INNER JOIN customers  ON customers.customerid = orders.customerid  
WHERE totalamount <= (SELECT max(totalamount) FROM orders)
Order by totalamount DESC 
limit 3



2)SELECT firstname, lastname, orderid, sum(totalamount) OVER (
PARTITION BY orders.totalamount 
ORDER BY totalamount DESC)
FROM orders INNER JOIN customers ON customers.customerid = orders.customerid
limit 3


3)SELECT firstname, lastname 
FROM customers INNER JOIN orders ON customers.customerid = orders.customerid
	WHERE totalamount > (SELECT avg(totalamount) FROM orders)
ORDER BY totalamount DESC
