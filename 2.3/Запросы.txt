1) SELECT firstname, lastname, totalamount FROM
customers INNER JOIN orders ON customers.customerid = orders.customerid
ORDER BY totalamount DESC;

2)SELECT customerid, firstname, lastname
FROM customers
WHERE customerid in (SELECT customerid FROM orders GROUP BY customerid ORDER BY SUM(totalamount) DESC LIMIT 1);


3)SELECT customers.customerid, customers.firstname, customers.lastname, orders.orderid, orders.totalamount
FROM orders inner join customers
on orders.customerid = customers.customerid
WHERE customers.customerid in (SELECT customerid FROM orders GROUP BY customerid ORDER BY SUM(totalamount) DESC LIMIT 1) 
ORDER BY totalamount DESC;
